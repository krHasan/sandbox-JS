save - Create


    function saveCategory() {
        if (!$("#formId").validationEngine('validate')) {
            return false;
        }

        let url = '/${params.controller}/save';

        SubmissionLoader.showTo()
        $.ajax({
            type: 'POST',
            data: $("#formId").serialize(),
            url: url,
            dataType: 'json',
            success: function (data, textStatus) {
                SubmissionLoader.hideFrom();
                MessageRenderer.render(data);
                table.ajax.reload();
                cancelCategory();
            },
            error: function (XMLHttpRequest, textStatus, errorThrown) {
            },
            complete: function () {
            }
        })
        return false;
    }

show - Read


    function onCategoryChange() {
        _categoryId = $('#categoryId').val();
        $.ajax({
            url: '/${params.controller}/getSubCategoryByCategoryId',
            type: 'post',
            data: {categoryId: _categoryId}
        }).done(function (json) {
            let tableReference = $('#tableId').dataTable();
            oSettings = tableReference.fnSettings();
            tableReference.fnClearTable(this);
            for (var i = 0; i < json.aaData.length; i++) {
                tableReference.oApi._fnAddData(oSettings, json.aaData[i]);
            }
            oSettings.aiDisplay = oSettings.aiDisplayMaster.slice();
            tableReference.fnDraw();
        }).fail(function (jqXHR, textStatus, errorThrown) {
        });
    }

update - Update


    function updateCategory() {
        if (!$("#formId").validationEngine('validate')) {
            return false;
        }

        let url = '/${params.controller}/update';

        $('#dialog').dialog('destroy');
        $('#dialogId').dialog({
            resizable: true,
            height: 200,
            width: 600,
            modal: true,
            title: 'Confirmation',
            buttons: {
                Update: function () {
                    $(this).dialog('close');
                    SubmissionLoader.showTo()
                    $.ajax({
                        type: 'PUT',
                        data: $('#formId').serialize(),
                        url: url,
                        dataType: 'json',
                        success: function (data, textStatus) {
                            SubmissionLoader.hideFrom();
                            MessageRenderer.render(data);
                            cancelCategory();
                            table.ajax.reload();
                        },
                        error: function (XMLHttpRequest, textStatus, errorThrown) {
                        },
                        complete: function () {
                        }
                    })
                },
                Cancel: function () {
                    $(this).dialog('close');
                    cancelCategory();
                }
            }
        });

        return false;
    }
