Input Decimal Number Only

Make a input field to take only decimal number value

Declaration

    function setDecimalField(inputFieldClass) {
		$("." + inputFieldClass).inputFilter(function(value) {
		    return /^[0-9]*\.?[0-9]*$/.test(value);
		});
	}
How to use
JS 

    $(document).ready(function () {
        setDecimalField('setDecimalField');
    });
GSP 

    use 'setDecimalField' this class name to class list

Precision Two in Decimal Number

Declaration

    function toFixedPrecisionTwo(id) {
        let toFixedPrecisionTwoValue = $("#" + id).val();
        if (toFixedPrecisionTwoValue != "" && toFixedPrecisionTwoValue != null 
                                        && toFixedPrecisionTwoValue != undefined) {
            toFixedPrecisionTwoValue = parseFloat(toFixedPrecisionTwoValue);
            $("#" + id).val(toFixedPrecisionTwoValue.toFixed(2));
        }
    }
How to use

    $(document).ready(function () {
        $("#id").blur(function () {
            toFixedPrecisionTwo('id');
        });
    });

Add Comma (,) in currency value

You can add comma in a number in below ways

By declaring a function

Declaration

    function numberAndCommaOnlyInputByClass(className) {
        $("." + className).inputFilter(function(value) {
            // Allow digits and comma only, using a RegExp
            return /^[0-9,]+$/g.test(value);    
        });
    }
How to use
Initialize in document ready

    $(document).ready(function () {
        numberAndCommaOnlyInputByClass('numberAndCommaOnly');
    });
Then use a class name in html elements

    class="numberAndCommaOnly"

Use JavaScript native support


    $("#totalAmount").val(_totalAmount.toLocaleString());
